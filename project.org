import (
    "github.com/jmoiron/sqlx"
    _ "github.com/jackc/pgx/v5/stdlib" // This registers "pgx" as a driver
)

func main() {
    // Connect using "pgx" as the driver name
    dsn := "postgres://user:pass@localhost:5432/todo_db"
    db, err := sqlx.Connect("pgx", dsn)
    if err != nil {
        panic(err)
    }
}
